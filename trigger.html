Question 10: AFTER DELETE - Copy to Audit Table

Write a trigger to copy deleted records into another table (audit table).

Create Audit Table:
CREATE TABLE emp_audit (
    audit_id NUMBER PRIMARY KEY,
    empno NUMBER,
    ename VARCHAR2(20),
    sal NUMBER,
    action VARCHAR2(10),
    audit_date DATE
);
Trigger Code:
CREATE OR REPLACE TRIGGER emp_delete_audit
AFTER DELETE ON emp
FOR EACH ROW
DECLARE
    v_id NUMBER;
BEGIN
    SELECT NVL(MAX(audit_id), 0) + 1 INTO v_id FROM emp_audit;
    INSERT INTO emp_audit VALUES (v_id, :OLD.empno, :OLD.ename, :OLD.sal, 'DELETE', SYSDATE);
    COMMIT;
    DBMS_OUTPUT.PUT_LINE('Deleted record archived: ' || :OLD.ename);
END emp_delete_audit;
Execution:
DELETE FROM emp WHERE empno = 9001;
Output:
Deleted record archived: ALEX

Question 20: Audit Trigger - Multi-Operation

Write a trigger to maintain audit table for INSERT, UPDATE, DELETE.

Trigger Code (Handles All DML):
CREATE OR REPLACE TRIGGER emp_full_audit
AFTER INSERT OR UPDATE OR DELETE ON emp
FOR EACH ROW
DECLARE
    v_action VARCHAR2(10);
    v_id NUMBER;
BEGIN
    SELECT NVL(MAX(audit_id), 0) + 1 INTO v_id FROM emp_audit;
    IF INSERTING THEN v_action := 'INSERT'; END IF;
    IF UPDATING THEN v_action := 'UPDATE'; END IF;
    IF DELETING THEN v_action := 'DELETE'; END IF;
    INSERT INTO emp_audit (audit_id, empno, action, audit_date, user_name)
    VALUES (v_id, :NEW.empno, v_action, SYSDATE, USER);
    DBMS_OUTPUT.PUT_LINE('Audit logged: ' || v_action);
END emp_full_audit;
Execution:
INSERT INTO emp (empno, ename, sal, deptno) VALUES (9009, 'AUDIT_TEST', 3000, 10);
Output:
Audit logged: INSERT

Question 27: Restrict Salary Update Beyond 20%

Write a trigger to restrict salary update beyond 20%.

BEFORE UPDATE OF sal ON emp
FOR EACH ROW

Question 30: Compound Trigger (Advanced)

Write a compound trigger (advanced Oracle feature).

Compound Trigger Code (Oracle 11g+):
CREATE OR REPLACE TRIGGER emp_compound
FOR INSERT OR UPDATE OR DELETE ON emp
COMPOUND TRIGGER
    v_row_count NUMBER := 0;
    v_action VARCHAR2(10);

    BEFORE STATEMENT IS
    BEGIN
        DBMS_OUTPUT.PUT_LINE('===== COMPOUND TRIGGER START =====');
    END BEFORE STATEMENT;

    BEFORE EACH ROW IS
    BEGIN
        v_row_count := v_row_count + 1;
        IF INSERTING THEN v_action := 'INSERT'; END IF;
        IF UPDATING THEN v_action := 'UPDATE'; END IF;
        IF DELETING THEN v_action := 'DELETE'; END IF;
        DBMS_OUTPUT.PUT_LINE('Processing row: ' || v_row_count || ' (' || v_action || ')');
    END BEFORE EACH ROW;

    AFTER STATEMENT IS
    BEGIN
        DBMS_OUTPUT.PUT_LINE('Total rows affected: ' || v_row_count);
        DBMS_OUTPUT.PUT_LINE('===== COMPOUND TRIGGER END =====');
    END AFTER STATEMENT;
END emp_compound;
Execution:
UPDATE emp SET sal = sal + 500 WHERE deptno = 10;
Output:
===== COMPOUND TRIGGER START =====
Processing row: 1 (UPDATE)
Processing row: 2 (UPDATE)
Processing row: 3 (UPDATE)
Total rows affected: 3
===== COMPOUND TRIGGER END =====
