CREATE TABLE error_log (
    log_id NUMBER PRIMARY KEY,

Question 27: Advanced Exception Handling - Nested Blocks with Re-raise

Write a PL/SQL block to implement nested exception blocks with re-raise mechanism.

PL/SQL Code:
DECLARE
    v_sal NUMBER := 5000;
    v_bonus_rate NUMBER := 50;
    v_bonus NUMBER;
BEGIN
    DBMS_OUTPUT.PUT_LINE('===== SALARY PROCESSING =====');
    BEGIN
        DBMS_OUTPUT.PUT_LINE('Level 1: Validating salary...');
        BEGIN
            DBMS_OUTPUT.PUT_LINE('Level 2: Calculating bonus...');
            IF v_bonus_rate > 30 THEN
                RAISE_APPLICATION_ERROR(-20100, 'Bonus rate exceeds limit');
            END IF;
            v_bonus := (v_sal * v_bonus_rate) / 100;
            DBMS_OUTPUT.PUT_LINE('Level 2: Bonus calculated = ' || v_bonus);
        EXCEPTION
            WHEN OTHERS THEN
                DBMS_OUTPUT.PUT_LINE('Level 2: Error - ' || SQLERRM);
                RAISE;
        END;
    EXCEPTION
        WHEN OTHERS THEN
            DBMS_OUTPUT.PUT_LINE('Level 1: Error caught - ' || SQLERRM);
            RAISE;
    END;
EXCEPTION
    WHEN OTHERS THEN
        DBMS_OUTPUT.PUT_LINE('Level 0 (Main): Final error - ' || SQLERRM);
END;
Output:
===== SALARY PROCESSING =====
Level 1: Validating salary...
Level 2: Calculating bonus...
Level 2: Error - ORA-20100: Bonus rate exceeds limit
Level 1: Error caught - ORA-20100: Bonus rate exceeds limit
