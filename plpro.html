(IN OUT Parameter)

PL/SQL Block:
CREATE OR REPLACE PROCEDURE raise_salary(p_empno IN NUMBER, p_increment IN OUT NUMBER) IS v_sal emp.sal%TYPE; BEGIN SELECT sal INTO v_sal FROM emp WHERE empno = p_empno; v_sal := v_sal + p_increment; UPDATE emp SET sal = v_sal WHERE empno = p_empno; p_increment := v_sal; COMMIT; END; / DECLARE v_new_sal NUMBER := 500; BEGIN raise_salary(7521, v_new_sal); DBMS_OUTPUT.PUT_LINE('New Salary: ' || v_new_sal); END; /

Question 31: Write a function to return nth highest salary.

(Function + Logic)

PL/SQL Block:
CREATE OR REPLACE FUNCTION nth_highest_salary(p_rank NUMBER) RETURN NUMBER IS v_salary emp.sal%TYPE; BEGIN SELECT sal INTO v_salary FROM ( SELECT sal, DENSE_RANK() OVER (ORDER BY sal DESC) rnk FROM emp ) WHERE rnk = p_rank AND ROWNUM = 1; RETURN v_salary; END; / BEGIN DBMS_OUTPUT.PUT_LINE('3rd highest salary: ' || nth_highest_salary(3)); END; /
Output:
3rd highest salary: 3000 PL/SQL procedure successfully completed.
Question 32: Write a procedure to implement transaction (COMMIT / ROLLBACK).

(Procedure + Transaction Control)

PL/SQL Block:
CREATE OR REPLACE PROCEDURE transfer_salary(p_from_emp NUMBER, p_to_emp NUMBER, p_amount NUMBER) IS v_from_sal emp.sal%TYPE; v_to_sal emp.sal%TYPE; BEGIN SELECT sal INTO v_from_sal FROM emp WHERE empno = p_from_emp FOR UPDATE; SELECT sal INTO v_to_sal FROM emp WHERE empno = p_to_emp FOR UPDATE; IF v_from_sal >= p_amount THEN UPDATE emp SET sal = v_from_sal - p_amount WHERE empno = p_from_emp; UPDATE emp SET sal = v_to_sal + p_amount WHERE empno = p_to_emp; COMMIT; DBMS_OUTPUT.PUT_LINE('Transfer completed.'); ELSE ROLLBACK; DBMS_OUTPUT.PUT_LINE('Transfer failed: insufficient funds.'); END IF; END; / BEGIN transfer_salary(7839, 7566, 500); END; /
Output:
Transfer completed. PL/SQL procedure successfully completed.
