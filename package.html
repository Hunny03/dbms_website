Question 5: Package Design

Create a package to group employee-related operations.

Package Specification:
CREATE OR REPLACE PACKAGE employee_ops AS
    PROCEDURE hire_employee(p_empno IN NUMBER, p_ename IN VARCHAR2, p_sal IN NUMBER);
    PROCEDURE fire_employee(p_empno IN NUMBER);
    FUNCTION get_employee_count RETURN NUMBER;
END employee_ops;
Package Body:
CREATE OR REPLACE PACKAGE BODY employee_ops AS
    PROCEDURE hire_employee(p_empno IN NUMBER, p_ename IN VARCHAR2, p_sal IN NUMBER) IS
    BEGIN
        INSERT INTO emp (empno, ename, sal) VALUES (p_empno, p_ename, p_sal);
        COMMIT;
        DBMS_OUTPUT.PUT_LINE('Employee hired successfully');
    END hire_employee;
    PROCEDURE fire_employee(p_empno IN NUMBER) IS
    BEGIN
        DELETE FROM emp WHERE empno = p_empno;
        COMMIT;
        DBMS_OUTPUT.PUT_LINE('Employee fired successfully');
    END fire_employee;
    FUNCTION get_employee_count RETURN NUMBER IS
        v_count NUMBER;
    BEGIN
        SELECT COUNT(*) INTO v_count FROM emp;
        RETURN v_count;
    END get_employee_count;
END employee_ops;
Execution:
BEGIN
    employee_ops.hire_employee(9999, 'JOHN', 3000);
    DBMS_OUTPUT.PUT_LINE('Total employees: ' || employee_ops.get_employee_count());
END;
Output:
Employee hired successfully
Total employees: 15

Question 19: Package + Cursor

Create a package that uses cursor to display all employees.

Package Specification:
CREATE OR REPLACE PACKAGE emp_cursor_pkg AS
    PROCEDURE display_all_employees;
END emp_cursor_pkg;
Package Body:
CREATE OR REPLACE PACKAGE BODY emp_cursor_pkg AS
    PROCEDURE display_all_employees IS
        CURSOR emp_cur IS SELECT empno, ename, job, sal FROM emp;
        v_empno emp.empno%TYPE;
        v_ename emp.ename%TYPE;
        v_job emp.job%TYPE;
        v_sal emp.sal%TYPE;
    BEGIN
        OPEN emp_cur;
        LOOP
            FETCH emp_cur INTO v_empno, v_ename, v_job, v_sal;
            EXIT WHEN emp_cur%NOTFOUND;
            DBMS_OUTPUT.PUT_LINE(v_empno || ' - ' || v_ename || ' - ' || v_job || ' - ' || v_sal);
        END LOOP;
        CLOSE emp_cur;
    END display_all_employees;
END emp_cursor_pkg;
Execution:
BEGIN
    emp_cursor_pkg.display_all_employees;
END;
Output:
7369 - SMITH - CLERK - 800
7499 - ALLEN - SALESMAN - 1600
7521 - WARD - SALESMAN - 1250
(... more records)

Question 25: Package for Audit Logs

Create a package to maintain employee audit logs.

Create Audit Table:
CREATE TABLE emp_audit (
    audit_id NUMBER PRIMARY KEY,
    empno NUMBER,
    action VARCHAR2(20),
    old_value VARCHAR2(100),
    new_value VARCHAR2(100),
    audit_date DATE
);
Package Specification:
CREATE OR REPLACE PACKAGE audit_pkg AS
    PROCEDURE log_action(p_empno IN NUMBER, p_action IN VARCHAR2, p_old_val IN VARCHAR2, p_new_val IN VARCHAR2);
    PROCEDURE display_audit_logs;
END audit_pkg;
Package Body:
CREATE OR REPLACE PACKAGE BODY audit_pkg AS
    PROCEDURE log_action(p_empno IN NUMBER, p_action IN VARCHAR2, p_old_val IN VARCHAR2, p_new_val IN VARCHAR2) IS
        v_id NUMBER;
    BEGIN
        SELECT NVL(MAX(audit_id), 0) + 1 INTO v_id FROM emp_audit;
        INSERT INTO emp_audit VALUES (v_id, p_empno, p_action, p_old_val, p_new_val, SYSDATE);
        COMMIT;
    END log_action;

Question 28: Business Logic Package

Create a package that calculates salary, bonus, and tax using multiple functions.

Package Specification:
CREATE OR REPLACE PACKAGE salary_calc_pkg AS
    FUNCTION calculate_gross_salary(p_basic IN NUMBER, p_da_pct IN NUMBER, p_hra_pct IN NUMBER) RETURN NUMBER;
    FUNCTION calculate_bonus(p_salary IN NUMBER, p_bonus_pct IN NUMBER) RETURN NUMBER;
    FUNCTION calculate_tax(p_salary IN NUMBER) RETURN NUMBER;
    FUNCTION calculate_net_salary(p_basic IN NUMBER, p_da_pct IN NUMBER, p_hra_pct IN NUMBER, p_bonus_pct IN NUMBER) RETURN NUMBER;
END salary_calc_pkg;
Package Body:
CREATE OR REPLACE PACKAGE BODY salary_calc_pkg AS
    FUNCTION calculate_gross_salary(p_basic IN NUMBER, p_da_pct IN NUMBER, p_hra_pct IN NUMBER) RETURN NUMBER IS
    BEGIN
        RETURN p_basic + (p_basic * p_da_pct / 100) + (p_basic * p_hra_pct / 100);
    END calculate_gross_salary;
    FUNCTION calculate_bonus(p_salary IN NUMBER, p_bonus_pct IN NUMBER) RETURN NUMBER IS
    BEGIN
        RETURN (p_salary * p_bonus_pct) / 100;
    END calculate_bonus;
    FUNCTION calculate_tax(p_salary IN NUMBER) RETURN NUMBER IS
    BEGIN
        IF p_salary > 5000 THEN RETURN (p_salary * 10) / 100;
        ELSIF p_salary > 3000 THEN RETURN (p_salary * 5) / 100;
        ELSE RETURN 0; END IF;
    END calculate_tax;
    FUNCTION calculate_net_salary(p_basic IN NUMBER, p_da_pct IN NUMBER, p_hra_pct IN NUMBER, p_bonus_pct IN NUMBER) RETURN NUMBER IS
        v_gross NUMBER; v_bonus NUMBER; v_tax NUMBER;
    BEGIN
        v_gross := calculate_gross_salary(p_basic, p_da_pct, p_hra_pct);
        v_bonus := calculate_bonus(v_gross, p_bonus_pct);
        v_tax := calculate_tax(v_gross + v_bonus);
        RETURN v_gross + v_bonus - v_tax;
    END calculate_net_salary;
END salary_calc_pkg;
Execution:
BEGIN
    DBMS_OUTPUT.PUT_LINE('Gross Salary: ' || salary_calc_pkg.calculate_gross_salary(10000, 10, 15));
    DBMS_OUTPUT.PUT_LINE('Bonus: ' || salary_calc_pkg.calculate_bonus(11500, 10));
    DBMS_OUTPUT.PUT_LINE('Tax: ' || salary_calc_pkg.calculate_tax(12650));
    DBMS_OUTPUT.PUT_LINE('Net Salary: ' || salary_calc_pkg.calculate_net_salary(10000, 10, 15, 10));
END;

Package Body:
CREATE OR REPLACE PACKAGE BODY multi_table_pkg AS
    PROCEDURE get_dept_employees(p_deptno IN NUMBER) IS
    BEGIN
        FOR rec IN (SELECT e.empno, e.ename, d.dname FROM emp e, dept d WHERE e.deptno = d.deptno AND e.deptno = p_deptno) LOOP
            DBMS_OUTPUT.PUT_LINE(rec.empno || ' - ' || rec.ename || ' (' || rec.dname || ')');
        END LOOP;
    END get_dept_employees;
    PROCEDURE get_dept_salary_info(p_deptno IN NUMBER) IS
    BEGIN
        FOR rec IN (SELECT d.dname, COUNT(e.empno) emp_count, SUM(e.sal) total_sal, AVG(e.sal) avg_sal FROM emp e, dept d WHERE e.deptno(+) = d.deptno AND d.deptno = p_deptno GROUP BY d.dname) LOOP
            DBMS_OUTPUT.PUT_LINE('Dept: ' || rec.dname || ' | Count: ' || rec.emp_count || ' | Total: ' || rec.total_sal || ' | Avg: ' || ROUND(rec.avg_sal, 2));
        END LOOP;

    FUNCTION get_dept_name(p_deptno IN NUMBER) RETURN VARCHAR2 IS

        DBMS_OUTPUT.PUT_LINE('Withdrawal of ' || p_amount || ' successful');
    END withdraw;
    FUNCTION check_balance(p_account_id IN NUMBER) RETURN NUMBER IS
        v_balance NUMBER;
    BEGIN
        SELECT balance INTO v_balance FROM bank_account WHERE account_id = p_account_id;
        RETURN v_balance;
    END check_balance;
END banking_pkg;

CREATE OR REPLACE PACKAGE BODY robust_pkg AS
    PROCEDURE insert_with_error_handling(p_empno IN NUMBER, p_ename IN VARCHAR2, p_sal IN NUMBER) IS
    BEGIN
        INSERT INTO emp (empno, ename, sal) VALUES (p_empno, p_ename, p_sal);
        COMMIT;
        DBMS_OUTPUT.PUT_LINE('Insert successful');
    EXCEPTION
        WHEN DUP_VAL_ON_INDEX THEN DBMS_OUTPUT.PUT_LINE('Error: Employee ID already exists');
        WHEN VALUE_ERROR THEN DBMS_OUTPUT.PUT_LINE('Error: Invalid data type');
        WHEN OTHERS THEN DBMS_OUTPUT.PUT_LINE('Error: ' || SQLERRM);
    END insert_with_error_handling;
    PROCEDURE update_with_error_handling(p_empno IN NUMBER, p_sal IN NUMBER) IS
    BEGIN
        UPDATE emp SET sal = p_sal WHERE empno = p_empno;
        IF SQL%ROWCOUNT = 0 THEN DBMS_OUTPUT.PUT_LINE('Error: Employee not found');
        ELSE COMMIT; DBMS_OUTPUT.PUT_LINE('Update successful');
        END IF;
    EXCEPTION WHEN OTHERS THEN DBMS_OUTPUT.PUT_LINE('Error: ' || SQLERRM);
    END update_with_error_handling;
    PROCEDURE delete_with_error_handling(p_empno IN NUMBER) IS
    BEGIN
        DELETE FROM emp WHERE empno = p_empno;
        IF SQL%ROWCOUNT = 0 THEN DBMS_OUTPUT.PUT_LINE('Error: Employee not found');
        ELSE COMMIT; DBMS_OUTPUT.PUT_LINE('Delete successful');
        END IF;
    EXCEPTION WHEN OTHERS THEN DBMS_OUTPUT.PUT_LINE('Error: ' || SQLERRM);
    END delete_with_error_handling;
END robust_pkg;
