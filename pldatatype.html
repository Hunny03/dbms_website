Question 8: Write a PL/SQL block to swap two numbers using variables.

(Variables)

PL/SQL Block:
DECLARE v_a NUMBER := 5; v_b NUMBER := 10; v_temp NUMBER; BEGIN DBMS_OUTPUT.PUT_LINE('Before Swap: a=' || v_a || ', b=' || v_b); v_temp := v_a; v_a := v_b; v_b := v_temp; DBMS_OUTPUT.PUT_LINE('After Swap: a=' || v_a || ', b=' || v_b); END; /
Output:

Question 13: Write a PL/SQL block to calculate simple interest using constant rate of interest.

(Constants)

PL/SQL Block:
DECLARE c_rate CONSTANT NUMBER := 5; -- 5% per annum v_principal NUMBER := 10000; v_time NUMBER := 2; v_interest NUMBER; BEGIN v_interest := (v_principal * c_rate * v_time) / 100; DBMS_OUTPUT.PUT_LINE('Principal: Rs.' || v_principal); DBMS_OUTPUT.PUT_LINE('Rate: ' || c_rate || '%'); DBMS_OUTPUT.PUT_LINE('Time: ' || v_time || ' years'); DBMS_OUTPUT.PUT_LINE('Simple Interest: Rs.' || v_interest); END; /
Output:
Principal: Rs.10000 Rate: 5% Time: 2 years Simple Interest: Rs.1000 PL/SQL procedure successfully completed.

Question 19: Write a PL/SQL block using variables, constants, and output statement together.

(Combined Concepts)

PL/SQL Block:
DECLARE c_company CONSTANT VARCHAR2(20) := 'TCS'; v_emp_name VARCHAR2(20) := 'RAVI'; v_basic_salary NUMBER := 50000; v_bonus NUMBER; BEGIN v_bonus := v_basic_salary * 0.15; DBMS_OUTPUT.PUT_LINE('=== SALARY SLIP ==='); DBMS_OUTPUT.PUT_LINE('Company: ' || c_company); DBMS_OUTPUT.PUT_LINE('Employee: ' || v_emp_name); DBMS_OUTPUT.PUT_LINE('Basic: ' || v_basic_salary); DBMS_OUTPUT.PUT_LINE('Bonus (15%): ' || v_bonus); DBMS_OUTPUT.PUT_LINE('Gross: ' || (v_basic_salary + v_bonus)); END; /
Output:
=== SALARY SLIP === Company: TCS Employee: RAVI Basic: 50000 Bonus (15%): 7500 Gross: 57500 PL/SQL procedure successfully completed.
Question 20: Write a PL/SQL block to display employee details using %ROWTYPE, constants, and comments.

(Combined Concepts)

PL/SQL Block:

Question 22: Write a PL/SQL block that uses all concepts: variable, constant, datatype, and output.

(Full Practice)

PL/SQL Block:
DECLARE -- Constants c_company_name CONSTANT VARCHAR2(30) := 'ABC CORPORATION'; c_company_code CONSTANT NUMBER := 100; -- Variables with different datatypes v_emp EMP%ROWTYPE; v_salary_increase NUMBER := 2000; v_new_salary NUMBER; v_joining_date DATE; v_experience_years NUMBER; BEGIN -- Fetch employee record using %ROWTYPE SELECT * INTO v_emp FROM EMP WHERE EMPNO = 7566; -- Calculate new salary v_new_salary := v_emp.SAL + v_salary_increase; -- Set joining date v_joining_date := v_emp.HIREDATE; -- Display comprehensive details DBMS_OUTPUT.PUT_LINE('**** ' || c_company_name || ' ****'); DBMS_OUTPUT.PUT_LINE('Company Code: ' || c_company_code); DBMS_OUTPUT.PUT_LINE(''); DBMS_OUTPUT.PUT_LINE('Employee Details:'); DBMS_OUTPUT.PUT_LINE('Name: ' || v_emp.ENAME); DBMS_OUTPUT.PUT_LINE('Emp ID: ' || v_emp.EMPNO); DBMS_OUTPUT.PUT_LINE('Designation: ' || v_emp.JOB); DBMS_OUTPUT.PUT_LINE('Current Salary: Rs.' || v_emp.SAL); DBMS_OUTPUT.PUT_LINE('Salary Increase: Rs.' || v_salary_increase); DBMS_OUTPUT.PUT_LINE('New Salary: Rs.' || v_new_salary); DBMS_OUTPUT.PUT_LINE('Joining Date: ' || v_joining_date); END; /
Output:
**** ABC CORPORATION **** Company Code: 100 Employee Details: Name: JONES Emp ID: 7566 Designation: MANAGER Current Salary: Rs.2975 Salary Increase: Rs.2000 New Salary: Rs.4975 Joining Date: 02-APR-81 PL/SQL procedure successfully completed.
