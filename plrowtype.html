PL/SQL SELECT INTO, %TYPE, %ROWTYPE

Question 1: Write a PL/SQL block to fetch employee name from EMP table for empno = 7369 and display it.

(SELECT INTO Statement)

PL/SQL Block:
DECLARE v_name VARCHAR2(20); BEGIN SELECT ENAME INTO v_name FROM EMP WHERE EMPNO = 7369; DBMS_OUTPUT.PUT_LINE('Employee Name: ' || v_name); END; /
Output:
Employee Name: SMITH PL/SQL procedure successfully completed.
Question 2: Write a PL/SQL block to retrieve salary of an employee and print it.

(%TYPE + SELECT INTO)

Output:
Salary of Employee 7521: Rs.1250 PL/SQL procedure successfully completed.
Question 10: Write a PL/SQL block to calculate annual salary using %TYPE variable.

(%TYPE + Calculation)

PL/SQL Block:
DECLARE v_salary EMP.SAL%TYPE; v_annual_salary EMP.SAL%TYPE; BEGIN SELECT SAL INTO v_salary FROM EMP WHERE EMPNO = 7654; v_annual_salary := v_salary * 12; DBMS_OUTPUT.PUT_LINE('Monthly Salary: Rs.' || v_salary); DBMS_OUTPUT.PUT_LINE('Annual Salary: Rs.' || v_annual_salary); END; /
Output:
Monthly Salary: Rs.1250 Annual Salary: Rs.15000 PL/SQL procedure successfully completed.
Question 11: Write a PL/SQL block to update salary using %TYPE variable.

(%TYPE + UPDATE)

PL/SQL Block:
DECLARE v_empno EMP.EMPNO%TYPE := 7934; v_new_salary EMP.SAL%TYPE := 1500; BEGIN UPDATE EMP SET SAL = v_new_salary WHERE EMPNO = v_empno; DBMS_OUTPUT.PUT_LINE('Salary updated for Employee ID: ' || v_empno); DBMS_OUTPUT.PUT_LINE('New Salary: Rs.' || v_new_salary); END; /
Output:
Salary updated for Employee ID: 7934 New Salary: Rs.1500 PL/SQL procedure successfully completed.

Question 16: Write a PL/SQL block to update salary using %ROWTYPE variable.

(%ROWTYPE + UPDATE)

PL/SQL Block:
DECLARE v_emp EMP%ROWTYPE; BEGIN SELECT * INTO v_emp FROM EMP WHERE EMPNO = 7788; DBMS_OUTPUT.PUT_LINE('Employee: ' || v_emp.ENAME); DBMS_OUTPUT.PUT_LINE('Old Salary: Rs.' || v_emp.SAL); v_emp.SAL := v_emp.SAL + 500; UPDATE EMP SET SAL = v_emp.SAL WHERE EMPNO = v_emp.EMPNO; DBMS_OUTPUT.PUT_LINE('New Salary: Rs.' || v_emp.SAL); DBMS_OUTPUT.PUT_LINE('Salary Updated Successfully'); END; /
Output:
Employee: SCOTT Old Salary: Rs.3000 New Salary: Rs.3500 Salary Updated Successfully PL/SQL procedure successfully completed.
Question 17: Write a PL/SQL block to insert a record using %ROWTYPE.

(%ROWTYPE + INSERT)

PL/SQL Block:
DECLARE v_emp EMP%ROWTYPE; BEGIN v_emp.EMPNO := 9999; v_emp.ENAME := 'NEWHIRE'; v_emp.JOB := 'ENGINEER'; v_emp.SAL := 4500; v_emp.DEPTNO := 30; INSERT INTO EMP (EMPNO, ENAME, JOB, SAL, DEPTNO) VALUES (v_emp.EMPNO, v_emp.ENAME, v_emp.JOB, v_emp.SAL, v_emp.DEPTNO); DBMS_OUTPUT.PUT_LINE('New Employee Inserted:'); DBMS_OUTPUT.PUT_LINE('EMPNO: ' || v_emp.EMPNO); DBMS_OUTPUT.PUT_LINE('ENAME: ' || v_emp.ENAME); DBMS_OUTPUT.PUT_LINE('Salary: Rs.' || v_emp.SAL); END; /
Output:
New Employee Inserted: EMPNO: 9999 ENAME: NEWHIRE Salary: Rs.4500 PL/SQL procedure successfully completed.
